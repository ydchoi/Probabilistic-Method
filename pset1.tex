\begin{question}[2]
\hfill
\begin{figure}[h!]
  \centering
    \includegraphics[width=1\textwidth]{pm-1-2.png}
\end{figure}
\end{question}
\begin{solution}
By union bound, we have
\eQb
P( \bigvee_{i=1}^{n} \{ A_i \text{ is monochromatic }\} ) 
&\leq& \sum_{i=1}^{m} P(\{A_i \text{ is monochromatic}\}) \\
&=& m \cdot 4^{1 - {n \choose 2}}.
\eQe
As $m < 4^{n-1}$ and $n \leq 2$, it follows that
\eQb
m \cdot 4^{1 - {n \choose 2}} &<& 4^{n-1} \cdot 4^{1 - {n \choose 2}}  \\
&=& 4^{\frac{n-n^2}{2}} < 1,
\eQe
which primarily grants us
\eQb
P( \bigvee_{i=1}^{n} \{ A_i \text{ is monochromatic }\} ) < 1. 
\eQe
Therefore, the complement event of $\bigvee_{i=1}^{n}$,
namely the event where no $A_i$ is monochromatic, has positive probability. Hence, we have
shown that there is a coloring of $\Omega$ by 4 colors so that no $A_i$ is monochromatic. 
\hfill $\qed$ 

\end{solution}
